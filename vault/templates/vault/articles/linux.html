find .                    # view all within (and below) this directory
find . -type d                # --"-- but dirs only
find . -type f -name "filename.ext"    # view files by name
find . -type f -iname "file*"        # view files that start with "file" (iname = case insensitive)
find . -type f -mmin -10        # view files created in the last ten minutes
find . -mmin +10            # view files created earlier than in the last ten minutes
find . -mmin +5 -10            # between 5 and ten mins ago...
find . -mtime -10            # view files created in the last ten days
-amin -10    -atime +10
find . -size +5M -10M            # view files larger than 5Mb and smaller than 10Mb
find . -empty                # view empty files
find . -perm 777            #find the crazy 777-permission files!

#change owner of all files in a folder w. find:
#find, folder_name, exec change-owner-func, username and group, placeholder ({}), end the statement (\; (OR +))
find <folder_name> -exec chown vulpes:vulpesgroup {} \;
# same thing, but with chmod for all the dirs:
find <folder_name> -type d -exec chmod 775 {} \;
#see all files in folder w. permission 775:
find <folder_name> -perm 775
#search with maxdepth 1; only in the present dir, not below.
find . -type f -name "*.jpg" -maxdepth 1

#HEALTHY PRAXIS!
    Try configuring unconfigured packages:

    sudo dpkg --configure -a

    Update the contents of the repositories

    sudo apt-get update

    Try to fix missing dependencies:

    sudo apt-get -f install

    Update all packages with new versions available:

    sudo apt-get full-upgrade

    Reinstall Ubuntu desktop:

    sudo apt-get install --reinstall ubuntu-desktop

    Remove unnecessary packages:

    sudo apt-get autoremove

    Delete downloaded packages already installed:

    sudo apt-get clean

    Reboot the system to see if the issue was resolved:

    sudo reboot

#WHEN IT GOES AWRY:
hold alt+sysrq and type in the sequence r e i s u b (busier backwards!!) leaving a few seconds per letter for the proces to finish
